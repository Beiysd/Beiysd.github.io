    # 一键复制

    # Html

    <div>
  <p id="copys">xxxxxx</p>
  <span onClick={() => this.copy("copys")} title="复制代码">
    &#9997;
  </span>
  {/**textarea:可以使文本保持原格式 */}
  <textarea id="area" />
</div>

    # JS

    /**
 * @name copy
 * @description 复制功能
 */
copy = (id) => {
  let text = document.getElementById(`${id}`).innerText;
  let area = document.getElementById("area");
  area.value = text; // 修改文本框的内容
  area.select(); // 选择对象
  document.execCommand("Copy");
};
